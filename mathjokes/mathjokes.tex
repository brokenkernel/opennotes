\subsubsection{Calculus}
\begin{equation}
	\begin{split}
		\frac{d}{dx}
		\frac{1}{x}
		=
		\frac{\cancel{d}}{\cancel{d}x}
		\frac{1}{x}
		=
		\frac{}{x}
		\frac{1}{x}
		=
		- \frac{1}{x^2}
	\end{split}
\end{equation}

\end{equation}

\section{1 = 2}

\begin{equation}
	\begin{split}
		b        &= a \\
		ab       &= a^2 \\
		ab - b^2 &= a^2 - b^2 \\
		b(a - b) &= (a + b)(a - b) \\
		b\cancel{(a - b)} &= (a + b)\cancel{(a - b)} \\
		b        &= a + b \\
		b        &= b + b \\
		b        &= 2b \\
		1        &= 2
	\end{split}
\end{equation}

\end{equation}
